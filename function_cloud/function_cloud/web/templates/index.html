Welcome to Function Cloud

Deploy your Python functions to the cloud with a simple decorator.

Simple Function Deployment

from function_cloud import FC

# Configure with your token
FC.configure(token="your_token")

# Decorate your function
@FC.function
def add(x, y):
    return x + y

# Call it locally
result = add(5, 10)
print(f"Result: {result}")  # Result: 15

# Get the cloud URL
url = add.get_url()
print(f"Function URL: {url}")
